102 void Md2Model::load_frame_mesh(int frame_num) {
103  if (frame_num < 0 || frame_num >= header_.num_frames) return;
104 
105  // Vertices for vertex buffer. (vertex, texcoord, normal)
106  std::vector<float> vertices;
107  // Indices for index buffer.
108  std::vector<uint32_t> indices;
109 
110  for (int i = 0; i < header_.num_triangles; ++i) {
111  for (int j = 0; j < 3; ++j) {
112  md2::frame_t* ptr_frame = &frames_[frame_num];
113  md2::vertex_t* ptr_vert =
114  &ptr_frame->vertices[triangles_[i].index_vert[j]];
115  md2::texcoord_t* ptr_texcoord =
116  &texcoords_[triangles_[i].index_st[j]];
117 
118  // Load vertices.
119  vertices.push_back(ptr_vert->vertex[0] * ptr_frame->scale[0] +
120  ptr_frame->translate[0]);
121  vertices.push_back(ptr_vert->vertex[1] * ptr_frame->scale[1] +
122  ptr_frame->translate[1]);
123  vertices.push_back(ptr_vert->vertex[2] * ptr_frame->scale[2] +
124  ptr_frame->translate[2]);
125 
126  // Load texcoords.
127  vertices.push_back(static_cast<float>(ptr_texcoord->s) /
128  static_cast<float>(header_.skinwidth));
129  vertices.push_back(static_cast<float>(ptr_texcoord->t) /
130  static_cast<float>(header_.skinheight));
131 
132  // Load normals.
133  vertices.push_back(-pre_normals_[ptr_vert->normal_index][0]);
134  vertices.push_back(-pre_normals_[ptr_vert->normal_index][1]);
135  vertices.push_back(-pre_normals_[ptr_vert->normal_index][2]);

138  indices.push_back(i * 3 + (2 - j));

157  std::vector<uint32_t> indices;

161  md2::frame_t* ptr_frame_a = &frames_[frame_a];
162  md2::frame_t* ptr_frame_b = &frames_[frame_b];

176  ptr_frame_a->translate[0];
177  vert_a[1] = ptr_vert_a->vertex[1] * ptr_frame_a->scale[1] +
178  ptr_frame_a->translate[1];
179  vert_a[2] = ptr_vert_a->vertex[2] * ptr_frame_a->scale[2] +
