5 void BodyRigidPhysics::init(glm::vec3 pos, glm::vec3 rotation, float angle) {
6  Vector3 position(pos.x, pos.y, pos.z);
7  Quaternion o = Quaternion::identity();
8 
9  angle = angle / (180 / PI_RP3D);
10 
11  float x = rotation.x * sin(angle / 2);
12  float y = rotation.y * sin(angle / 2);
13  float z = rotation.z * sin(angle / 2);
14  float w = cos(angle / 2);
15 
16  float normal = sqrt(pow(cos(angle / 2), 2) +
17  pow(rotation.x, 2) * pow(sin(angle / 2), 2) +
18  pow(rotation.y, 2) * pow(sin(angle / 2), 2) +
19  pow(rotation.z, 2) * pow(sin(angle / 2), 2));
20 
21  o.setAllValues(x / normal, y / normal, z / normal, w / normal);
22  Transform t(position, o);
23  rb = Physics::getPhysicsWorld()->createRigidBody(t);
24  setType(2);
25 
26  created = true;
27 }
